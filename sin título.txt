<script>
    // Hora de envío de la carta (6:00 PM)
    var hora_envio = new Date();
    hora_envio.setHours(18, 0, 0, 0);  // Ajustar la hora a las 6:00 PM

    // Hora de objetivo: 12:00 AM del día siguiente
    var hora_objetivo = new Date(hora_envio);
    hora_objetivo.setHours(24, 0, 0, 0); // Establecer la hora de objetivo a las 12:00 AM

    // Calcular la cuenta regresiva
    var countdownDate = hora_objetivo.getTime();

    var countdownFunction = setInterval(function() {
        var now = new Date().getTime();
        var distance = countdownDate - now;

        var days = Math.floor(distance / (1000 * 60 * 60 * 24));
        var hours = Math.floor((distance % (1000 * 60 * 60 * 24)) / (1000 * 60 * 60));
        var minutes = Math.floor((distance % (1000 * 60 * 60)) / (1000 * 60));
        var seconds = Math.floor((distance % (1000 * 60)) / 1000);

        document.getElementById("countdown").innerHTML = "Cuenta regresiva: " + days + "d " + hours + "h "
        + minutes + "m " + seconds + "s ";

        if (distance < 0) {
            clearInterval(countdownFunction);
            document.getElementById("countdown").innerHTML = "¡Es momento de descargar el PDF!";
            document.getElementById("downloadBtn").style.display = "block"; // Mostrar el botón
        }
    }, 1000);

    // Función para descargar el PDF
    function downloadPDF() {
        // Aquí pones tu URL de descarga directa del archivo PDF
        var pdfUrl = 'https://drive.google.com/file/d/1PwYMGK5QC_C7HN543b8Ol7CLaySSMYbH/view?usp=drivesdk'; // Reemplaza con tu URL de Google Drive

        // Crear un enlace invisible y simular un clic para la descarga
        var link = document.createElement('a');
        link.href = pdfUrl;
        link.download = 'carta_personalizada.pdf'; // El nombre del archivo cuando se descargue
        link.click();
    }
</script>
